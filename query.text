1. SELECT COUNT(id) AS totale, YEAR(`enrolment_date`) AS year 
    FROM students GROUP BY year 

2. SELECT COUNT(id), `office_address`
    FROM `teachers`  
    GROUP by `office_address`   

3. SELECT AVG(`vote`) AS media_voti , `exam_id` 
    FROM `exam_student` GROUP BY exam_id

4. SELECT COUNT(`department_id`), department_id 
    FROM `degrees` GROUP by `department_id`
